def label_segment(row):

        return '5+7' # 완료일수가 짧고, 완료율이 낮고, 결제 금액이 낮음

        return '6+8' # 완료일 수가 길고, 완료율이 높고, 결제 금액이 높음
    else:
        return '기타'
